import { CommandHandler, ICommandHandler } from '@nestjs/cqrs';
import { UserNoneService } from './user-none.service';
import { UserNoneDto } from './user-none.dto';
import { UserId } from '../../domain/user-id';
import { UserName } from '../../domain/user-name';

@CommandHandler(UserNoneDto)
export class UserNoneHandler implements ICommandHandler<UserNoneDto> {
  constructor(private service: UserNoneService) {}

  async execute(dto: UserNoneDto): Promise<void> {
    const id = new UserId(dto.id);
    const name = new UserName(dto.name);

    await this.service.execute(id, name);
  }
}
