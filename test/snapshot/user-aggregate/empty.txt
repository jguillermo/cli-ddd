import { AggregateRoot } from 'base-ddd';
import { UserId } from './user-id';
import { UserName } from './user-name';

export class User extends AggregateRoot {
  constructor(private _id: UserId, private _name: UserName) {
    super();
  }

  get id(): UserId {
    return this._id;
  }

  get name(): UserName {
    return this._name;
  }
}
